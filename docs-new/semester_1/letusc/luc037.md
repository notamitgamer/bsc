#include <stdio.h>
#include <math.h>

double series(double x) // made this fn only for fun, making a fn was not necessary
{
    double result = (x - 1) / x;
    int i;
    for (i = 2; i <= 7; i++)
    {
        result += 0.5 * pow(((x - 1) / x), i);
    }
    return result;
}

int main()
{
    double x;
    printf("Enter the value for x : ");
    scanf("%lf", &x);
    printf("\nResult : %g", series(x));
    return 0;
}
