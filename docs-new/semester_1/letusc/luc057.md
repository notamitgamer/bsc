```c [luc057.c]
#include <stdio.h>
#include <math.h>
#include <stdlib.h>

int main()
{
    int arr[25], i, min;
    int *ptr;

    printf("Enter 25 integers:\n");
    for (i = 0; i < 25; i++)
    {
        scanf("%d", &arr[i]);
    }

    ptr = arr; // Point to start of array
    min = *ptr; // Initialize min with first element

    for (i = 1; i < 25; i++)
    {
        ptr++; // Move pointer to next element
        if (*ptr < min)
        {
            min = *ptr;
        }
    }

    printf("\nSmallest number is: %d\n", min);

    return 0;
}
```
