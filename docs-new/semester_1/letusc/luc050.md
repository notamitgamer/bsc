#include <stdio.h>
#include <math.h>
#include <stdlib.h>

int get_sum(int);

int main()
{
    int n = 25, sum;

    sum = get_sum(n);
    printf("Sum of first %d natural numbers is: %d\n", n, sum);

    return 0;
}

int get_sum(int n)
{
    if (n == 0)
        return 0;
    else
        return n + get_sum(n - 1);
}
