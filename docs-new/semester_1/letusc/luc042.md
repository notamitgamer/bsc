```c [luc042.c]
#include <stdio.h>
int main()
{
    int starter = 1;
    int tab = 3;
    printf("The pattern : \n");
    for (int i = 1; i <= 4; i++)
    {
        int count = 0;
        for (int k = 1; k <= tab; k++)
        {
            printf("    ");
        }
        tab = tab - 1;
        for (int j = starter; j <= 10; j++)
        {
            if (count >= i)
            {
                break;
            }
            printf("%d       ", j);
            count++;
            starter++;
        }
        printf("\n");
    }
    return 0;
}
```
