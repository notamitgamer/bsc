```c [luc109.c]
#include <stdio.h>
#include <stdlib.h>

int main()
{
    unsigned char num, res;

    printf("Enter an 8-bit number: ");
    scanf("%hhu", &num);

    // Odd bits usually refer to positions 1, 3, 5, 7.
    // Mask: 1010 1010 (binary) = 0xAA
    // OR operation sets bits to 1.
    
    res = num | 0xAA;

    printf("Original: 0x%02X\n", num);
    printf("Result:   0x%02X\n", res);

    return 0;
}
```
