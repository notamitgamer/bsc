#include <stdio.h>

int leapYear(int);

int leapYear(int year)
{
    if ((year % 400 == 0) || ((year % 4 == 0) && (year % 100 != 0)))
        return 1;
    else
        return 0;
}

int main()
{
    int year;
    printf("Enter the year : ");
    scanf("%d", &year);
    if (leapYear(year))
        printf("\nYear %d is a Leap Year.", year);
    else
        printf("\nYear %d is not a Leap Year.", year);
    return 0;
}
