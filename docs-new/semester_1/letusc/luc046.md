!!! abstract "Problem Statement"

```c
#include <stdio.h>

void circularShift(int *, int *, int *);

int main()
{
    int x = 5, y = 8, z = 10;

    printf("--- Before Shift ---\n");
    printf("x: %d, y: %d, z: %d", x, y, z);

    circularShift(&x, &y, &z);

    printf("\n--- After Shift ---\n");
    printf("x: %d, y: %d, z: %d", x, y, z);

    return 0;
}

void circularShift(int *x, int *y, int *z)
{
    int temp = *z;
    *z = *y;
    *y = *x;
    *x = temp;
}
```
