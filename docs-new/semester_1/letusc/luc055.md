```c [luc055.c]
#include <stdio.h>
#include <math.h>
#include <stdlib.h>

int main()
{
    int arr[25], i;
    int pos = 0, neg = 0, even = 0, odd = 0;

    printf("Enter 25 integers:\n");
    for (i = 0; i < 25; i++)
    {
        scanf("%d", &arr[i]);

        if (arr[i] > 0)
            pos++;
        else if (arr[i] < 0)
            neg++;

        if (arr[i] % 2 == 0)
            even++;
        else
            odd++;
    }

    printf("\nResults:\n");
    printf("Positive: %d\n", pos);
    printf("Negative: %d\n", neg);
    printf("Even: %d\n", even);
    printf("Odd: %d\n", odd);

    return 0;
}
```
