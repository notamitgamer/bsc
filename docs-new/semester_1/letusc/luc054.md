#include <stdio.h>
#include <math.h>
#include <stdlib.h>

#include "interest.h"

int main()
{
    float p, r, t, si, amt;

    printf("Enter Principal, Rate and Time: ");
    scanf("%f %f %f", &p, &r, &t);

    si = SIMPLE_INTEREST(p, r, t);
    amt = AMOUNT(p, si);

    printf("Simple Interest: %.2f\n", si);
    printf("Total Amount: %.2f\n", amt);

    return 0;
}
