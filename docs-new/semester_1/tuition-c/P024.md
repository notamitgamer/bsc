#include <stdio.h>
int main()
{
    int year;
    printf("Enter the year : ");
    scanf("%d", &year);
    if (year % 4 == 0 && year % 100 != 0)
        printf("\nYear %d is a leapyear.", year);
    else if (year % 400 == 0)
        printf("\nYear %d is a leapyear (Century).", year);
    else
        printf("\nYear %d is not a leapyear.", year);
    return 0;
}
