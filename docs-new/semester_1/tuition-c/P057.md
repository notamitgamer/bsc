## Source Code
<div data-search-exclude="true">
```c
#include <stdio.h>

void series(int n)
{
    int i;
    printf("\nSeries:  ");
    for (i = 1; i <= n; i++)
        printf("%d  ", (i * i * i) - 1);
}

int main()
{
    int n;
    printf("Enter n: ");
    scanf("%d", &n);
    series(n);
    return 0;
}
```
