#include<stdio.h>
int main() {
	double mrp, final_price, temp;
	printf("Enter the printed price : ");
	scanf("%lf", &mrp);
	temp = mrp * 0.90;
	final_price = temp * 1.06;
	printf("\nCustomer have to pay : %lf", final_price);
	return 0;
}
