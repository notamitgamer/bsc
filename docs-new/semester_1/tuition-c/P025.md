```c [P025.c]
#include <stdio.h>
int main()
{
    double unit, amt;
    printf("Enter the electricity consp(unit) : ");
    scanf("%lf", &amt);
    if (unit <= 50)
        amt = unit * 0.50;
    else if (unit > 50 && unit <= 50)
        amt = 25 + ((unit - 50) * 0.75);
    else if (unit > 150 && unit <= 250)
        amt = 25 + 75 + ((unit - 150) * 1.20);
    else if (unit > 250)
        amt = 25 + 75 + 120 + ((unit - 250) * 1.50);
    amt = amt * 1.20;
    printf("\nAmount to be paid : %g", amt);
    return 0;
}
```
