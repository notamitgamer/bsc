```c [P051.c]
#include <stdio.h>

int main()
{
    int a, b, i, sa = 0, sb = 0;
    printf("Enter two number : ");
    scanf("%d %d", &a, &b);
    for (i = 1; i <= a / 2; i++)
        if (a % i == 0)
            sa += i;
    for (i = 1; i <= b / 2; i++)
        if (b % i == 0)
            sb += i;
    if (sa == b && sb == a)
        printf("\nInput %d and %d is Amicable Pair.", a, b);
    else
        printf("\nInput %d and %d is Not Amicable Pair.", a, b);
    return 0;
}
```
