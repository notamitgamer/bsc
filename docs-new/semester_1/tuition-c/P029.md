#include <stdio.h>
int main()
{
    int num, i = 0, result = 0;
    printf("Enter the value for n : ");
    if (scanf("%d", &num) != 1)
    {
        printf("\nPlease enter a number.");
        return 1;
    }
    if (num < 1)
    {
        printf("\nPlease enter a positive number.");
        return 1;
    }
    while (i <= num)
    {
        result = result + i;
        i++;
    }
    printf("\nResult : %d", result);
    return 0;
}
