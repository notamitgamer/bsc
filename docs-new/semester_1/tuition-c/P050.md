!!! abstract "Problem Statement"

#include <stdio.h>
#include <math.h>

int main()
{
    int num, res = 0, temp, position = 0;
    printf("Enter the number : ");
    scanf("%d", &num);
    temp = num;
    while (temp > 0)
    {
        position++;
        temp /= 10;
    }
    temp = num;
    while (temp > 0)
    {
        res += (int)pow(temp % 10, position);
        position--;
        temp /= 10;
    }
    if (res == num)
        printf("\nInput %d is a Disarium Number.", num);
    else
        printf("\nInput %d is Not a Disarium Number.", num);
    return 0;
}
