## Source Code

```c
#include <stdio.h>
int main()
{
    double dis, amt;
    printf("Enter the distance : ");
    scanf("%lf", &dis);
    if (dis <= 5.0)
        amt = 90.0;
    else if (dis > 5.0 && dis <= 15.0)
        amt = 90.0 + (dis - 5.0) * 20;
    else if (dis > 15.0 && dis <= 25.0)
        amt = 90.0 + 200.0 + (dis - 15.0) * 10;
    else if (dis > 25.0)
        amt = 90.0 + 200.0 + 100.0 + (dis - 25.0) * 9;
    printf("\nAmount to be paid : %g", amt);
    return 0;
}
```
