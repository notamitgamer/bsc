## Source Code
```c
#include <stdio.h>

int main()
{
    int n, sum = 0, temp = 0, i;
    printf("Enter the n : ");
    scanf("%d", &n);
    for (i = 1; i <= n; i++)
    {
        temp += i;
        sum += temp;
    }
    printf("%d", sum);
    return 0;
}
```
