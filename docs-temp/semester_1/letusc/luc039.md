!!! abstract "Problem Statement"

```c
#include <stdio.h>
int main()
{
    int i, population = 100000;
    printf("Present year : %d\n", population);
    for (i = 1; i <= 10; i++)
    {
        population /= 1.10;
        printf("%d year ago : %d\n", i, population);
    }
    return 0;
}
```
