#include <stdio.h>
#include <math.h>
#include <stdlib.h>

int main()
{
    int arr[100], n, i, symmetric = 1;

    printf("Enter number of elements (n): ");
    scanf("%d", &n);

    printf("Enter %d elements: ", n);
    for (i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }

    // Check symmetry
    for (i = 0; i < n / 2; i++)
    {
        if (arr[i] != arr[n - 1 - i])
        {
            symmetric = 0;
            break;
        }
    }

    if (symmetric)
        printf("\nThe array is symmetric (Palindrome).\n");
    else
        printf("\nThe array is NOT symmetric.\n");

    return 0;
}
