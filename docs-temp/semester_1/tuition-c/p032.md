```c
#include <stdio.h>
int main()
{
    int n, temp, sod = 0;
    printf("Enter the number : ");
    scanf("%d", &n);
    temp = n;
    while (temp > 0)
    {
        sod = sod + (temp % 10);
        temp = temp / 10;
    }
    if (n % sod == 0)
        printf("\nIt is Niven number.");
    else
        printf("\nIt is not a Niven number.");
    return 0;
}
```
