```c
#include <stdio.h>
int main()
{
    double bs, gs, ns, da, hra, pf;
    printf("Enter the basic salary of the employee : ");
    scanf("%lf", &bs);
    da = bs * 0.25;
    hra = bs * 0.125;
    pf = bs * 0.10;
    gs = bs + da + hra;
    ns = gs - pf;
    printf("\nGross Salary : %g"
           "\nNet Salary   : %g",
           gs, ns);
    return 0;
}
```
