#include <stdio.h>
int main()
{
    int a, b, c, min;
    printf("Enter three number : ");
    scanf("%d %d %d", &a, &b, &c);
    min = (a < b && a < c) ? a : (b < a && b < c) ? b
                                                  : c;
    printf("Minimum = %d", min);
    return 0;
}
