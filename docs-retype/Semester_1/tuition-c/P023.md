#include <stdio.h>
int main()
{
    double cost, amt;
    printf("Enter the total cost : ");
    scanf("%lf", &cost);
    if (cost <= 2000)
        amt = cost * 0.94;
    else if (cost > 2000 && cost <= 5000)
        amt = cost * 0.9;
    else if (cost > 5000 && cost <= 10000)
        amt = cost * 0.85;
    else if (cost > 10000)
        amt = cost * 0.8;
    printf("\nAmount to be paid : %g", amt);
    return 0;
}
