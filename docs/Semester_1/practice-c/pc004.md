#include <stdio.h>
int main()
{
    int i, j, k, n;
    printf("Enter n : ");
    scanf("%d", &n);
    for (i = n; i >= 1; i--)
    {
        for (j = 1; j <= i - 1; j++)
        {
            printf("\t");
        }
        for (k = i; k >= 1; k--)
        {
            printf("%d\t", k);
        }
        printf("\n");
    }
    return 0;
}
