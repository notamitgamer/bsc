## Source Code
```c
#include <stdio.h>

int main()
{
    int a, b, temp, temp_a, temp_b;
    printf("Enter the a and b : ");
    scanf("%d %d", &a, &b);
    temp_a = a, temp_b = b;
    while (b > 0)
    {
        temp = a;
        a = b;
        b = temp % b;
    }
    printf("\nHCF of %d and %d is : %d\n", temp_a, temp_b, a);
    return 0;
}
```
