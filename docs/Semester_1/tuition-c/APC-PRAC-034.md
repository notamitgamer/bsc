#include <stdio.h>
#include <math.h>

int isPrime(int);

int isPrime(int n)
{
    if (n <= 1)
        return 0;
    if (n == 2)
        return 1;
    if (n % 2 == 0)
        return 0;
    int temp = (int)sqrt(n), i;
    for (i = 3; i <= temp; i += 2)
        if (n % i == 0)
            return 0;
    return 1;
}

int main()
{
    int lb, ub, i;
    printf("Enter the lower bound and the upper bound: ");
    scanf("%d %d", &lb, &ub);
    printf("\nPrime numbers between %d and %d:  ", lb, ub);
    for (i = lb; i <= ub; i++)
        if (isPrime(i))
            printf("%d  ", i);
    return 0;
}
