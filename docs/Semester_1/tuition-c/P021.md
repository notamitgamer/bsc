```c
#include <stdio.h>
#include <math.h>
int main()
{
    int n, temp;
    double sr;
    printf("Enter the number : ");
    scanf("%d", &n);
    sr = sqrt(n);
    temp = (int)sr;
    if (temp * temp == n)
        printf("\nThis is a perfect square.");
    else
        printf("\nThis is not a perfect square.");
    return 0;
}
```
